function handleGensel(_this) {
    ongoingGensel = true;
    const genName = $(_this).find('.f-pokolenie-name').text();
    const currYear = new Date().getFullYear();
    const genYears = $(_this).find('.f-pokolenie-years').text();
    const years = genYears.split(' - ');
    const yearFrom = years[0];
    const yearTo   = years[1] == 'н.в.' ? currYear : years[1];
    const genParent = document.getElementById('id-dd3');
    genParent.firstElementChild.innerText = genName + ' ' + genYears;
    $('#id_gen').val(_this.dataset.itemId);
    $('#slug_gen').val(_this.dataset.slug);
    // Сбрасываем выбор года
    const yearLabel = $('#id-dd4').children().first();
    yearLabel.text(yearLabel.attr('data-default-text'));
    $('#year_ot').val(0);
    $('#year_do').val(0);
    
    // Ограничиваем диапазон выбора года
    $.getJSON("/aj/filter.php?a=complect_sel&is_new=0&gen="+_this.dataset.itemId+'&location_id=' + locationId, function(data){
        if (data.yearsF.length) {            
            $('.yearser > li').each(function() {
                if (0 == this.value) return;
                if (data.yearsF.includes(this.value.toString())) {
                    this.style.display = 'list-item';
                } else {
                    this.style.display = 'none';
                }  
            });
        } else {
            $('.yearser > li').each(function() {
                if (0 == this.value) return;
                this.style.display = 'list-item';
            });
        }
        // console.log(data);
        $('.dd4').removeClass('dd-disabled');
    });
    loadMainBillsCount();
    ongoingGensel = false;
}

function loadMainBillsCount() {
    let postData = $("#dd_form").serialize();
    $.post("/aj/search.php?action=count_bills", postData, function(data){
        const response = JSON.parse(data);
        if (response) {
            $('#f-sf-counter').removeClass('loading');
            if (response.total_count > 0) {
                $('#dd-submit').removeClass('dd-empty-set');
                $('#f-sf-msg').text('Показать');
                $('#f-sf-counter').text(response.total_count);
                $('#f-sf-noun').text(declOfNum(response.total_count,['объявление', 'объявления', 'объявлений']));
            } else {                
                $('#f-sf-counter').text('  ');
                $('#f-sf-msg').text('Ничего');
                $('#f-sf-noun').text('не найдено');
                $('#dd-submit').addClass('dd-empty-set');

            }
        }
    });
}
